% THEOREM
\newtcolorbox[
    % auto counter,number within=section
    ]%
    {theorem}[2][]{
    colback=white,
    colbacktitle=burntorange,
    colframe=burntorange!30!white,
    arc=0mm,
    fonttitle=\color{white},
    enhanced,
    attach boxed title to top left={yshift=-2mm,xshift=2mm},
    title=\textsc{Theorem:}~#2,
    #1
    }
